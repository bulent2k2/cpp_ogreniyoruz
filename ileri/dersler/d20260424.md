Adres değişkenleri, iki boyutlu diziler...

```c++
/******************************************************************************

Pointer (Adres değişkeni)  Gösterge
Bellek 
Iterator: Yineleyici (akıllı pointer)

   
1  2  3  4
          s
[ ][ ][4][10][ ]
       a

Turing Machine: upuzun serit, karelere bolunmus: gez, oku, yaz, 
Von Neumann: Neumann architecture.. CPU/RAM/DISK

0 .... 64GB -> 64/8 milyar int (bir int 64 bit = 8 bayt ise), aslinda milyardan fazla, 1Kbit = 1024bit (2^10), 1Mbit = 2^20, 1Gbit = 2^30
[ ][ ][4][5][ ][ ][ ][4][10][ ][ ][ ][4][10][ ][ ][ ][4][10][ ][ ][ ][4][10][ ][ ][ ][4][10][ ][ ][ ][4][10][ ]

2 3 4
4 5 6

S d[2][3] // d 

d[0], d[1]
d[0] -> [2 3 4] d[1] -> [4 5 6]

d?

d[0]      d[1]
 |        |
 v        v
[2 3 4]  [4 5 6]
[2 4] [3 5] [4 6]

*******************************************************************************/

#include <iostream>

using S = int;
using K = S;
using Adres = S *; // * turden tur ureten bir islemci. deger turunu adres turune ceviriyor
void yaz(Adres);
void dene1() {
    S s = 10, t = 20;
    Adres a = &s; // s'nin bellekte yazildigi yer, a'nin degeri adres. 
    Adres *aa = &a; // adres degiskeninin adresi
    std::cout << aa << " *aa= " << *aa << '\n';
    a = &t;
    std::cout << &a << " *a= " << *a << '\n';
    yaz(a);
}
void yaz(Adres a) {
  std::cout << &a << '\n'  // gostergenin bellekte yazildigi adres
	    << a << '\n'   // gostergenin gosterdigi yani tuttugu adres
	    << *a << '\n'; // o adresteki degiskenin degeri
}
void dene2() {
  S d1[2][3] { {2, 3, 4}, {4, 5, 6} };
  S d2[3][2] { {2, 4}, {3, 5}, {4, 6} };
  for(K x=0; x<2; ++x) {
    for(K y=0; y<3; ++y) 
      std::cout << d1[x][y] << " ";
    std::cout << std::endl;
  }

  for(K x=0; x<3; ++x) {
    for(K y=0; y<2; ++y) 
      std::cout << d2[x][y] << " ";
    std::cout << std::endl;
  }

  yaz(d1[0]);
  yaz(d1[1]);

  yaz(d2[0]);
  yaz(d2[1]);
  yaz(d2[2]);
}

int main() {
    dene1();
    dene2();
    return 0;
}

```
